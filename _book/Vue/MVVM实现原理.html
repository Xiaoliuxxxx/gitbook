1.MVVM实现原理

MVVM双向绑定原理

HTML代码如下：

<body>
    <div id="app">
        <h1>数据双向绑定</h1>
        <div v-text="myText"></div>
        <div v-text="myGod"></div>
        <input type="text" v-model="myText">
    </div>
    <script>
        const app = new Vue({
            el:"#app",
            data:{
                myText:"测试文本",
                myGod:"test text"
            }
        })
    </script>
</body>

MVVM双向绑定的思路大概如下：

"v-text"对于浏览器来说是未知的属性，无法得到解析，需要写一个解析函数，同时通过订阅者模式，在解析的同时收集有那些dom元素订阅了myText这个数据，即添加订阅者；在数据更新时通过数据劫持重写set方法，在数据更新之后通知订阅者进行update;

1.M-->V

通过Compile()函数来解析自定义的属性，并为每个数据收集订阅者

Compile(el){
    let nodes = el.children;//获取el下所有子元素

    for(let i=0;i<nodes.length;i++){
        let node = nodes[i];
        if(node.length){
            this.Compile(node);
        }//递归子元素的子元素
        if(node.hasAttribute('v-text')){
            let attribute = node.getAttribute('v-text');
            attribute && this._directives[attribute].push(new Watcher(node,this,attribute,'innerHTML'));
        }
        if(node.hasAttribute('v-model')){
            let _this = this;
            let attribute = node.getAttribute('v-model');
            attribute && this._directives[attribute].push(new Watcher(node,this,attribute,'value'));
            node.addEventListener('input',(function(){
                return function(){
                    console.log(this);
                    _this.$data[attribute] = node.value;
                    console.log(_this.$data[attribute]);
                }
            })());
        }
    }
}

2.V-->M

通过数据劫持重写set方法 在触发数据更新时通知观察者进行更新

Observer(data){
    for(let key in data){
        this._directives[key] = [];//初始化订阅者容器数组
        // myText:[],myGod:[]

        let Val = this.$data[key];
        let  _obj = this._directives[key];
        Object.defineProperty(this.$data,key,{

            get:function(){
                return Val;
            },
            set:function(newVal){
                if(Val !== newVal){
                    Val = newVal;
                    console.log(_obj);
                    _obj.forEach(element => {
                        element.update();
                        console.log(element.update)
                    });
                }
            }
        })
    }
}

3.watcher类

class Watcher{
    constructor(el, vm, exp, attr){
        this.el = el;
        this.vm = vm;
        this.exp = exp;
        this.attr = attr;
        this.update();
    }
    update(){
        this.el[this.attr] = this.vm.$data[this.exp];
    };
}
